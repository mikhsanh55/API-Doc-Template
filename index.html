http.setRequestHeader('Authorization', 'bearer YOUR_TOKEN')
http.overrideMimeType('application/json')

http.onload = function() {
  console.log(http.responseText)
}

http.onerror = function() {
  console.error(http.responseText)
}

http.send(null)

Detail Tugas


# Here is a curl example JS XMLHttp
var http = new XMLHttpRequest(),
    formData = new FormData()

http.open('POST', 'https://api.lms-sman5bdg.online/tugas/get-detail-tugas')
http.setRequestHeader('Authorization', 'bearer YOUR_TOKEN')
http.overrideMimeType('application/json')

formData.append('tugas', this.idTugas)

http.onload = function() {
  console.log(http.responseText)
}

http.onerror = function() {
  console.error(http.responseText)
}

http.send(formData)

To get detail of tugas siswa, you need to POST call to the following URL :
http://api.lms-sman5bdg.online/tugas/get-detail-tugas 



Result example 200:

{
  status: true,
  data: [
    detail: {

      nama_mapel: 'Matematika'
    },
    files: [
      {
        id: 1,
        id_tugas: 23,
        format: 'pdf',
        path: 'https://api.lms-sman5bdg.online/storage/tugas/attachment/file 1.pdf'
      }
    ]
  ]
}

POST QUERY PARAMETERS

Field Type Description
Header Authorization (mandatory) String bearer Your TOKEN.
Body tugas (mandatory) Integer Id tugas

Insert Assignment Tugas (as Siswa)


# Here is a curl example JS XMLHttp
var http = new XMLHttpRequest(),
    formData = new FormData()

http.open('POST', 'https://api.lms-sman5bdg.online/tugas/insert-assignment')
http.setRequestHeader('Authorization', 'bearer YOUR_TOKEN')
http.overrideMimeType('application/json')

formData.append('tugas', this.idTugas)
formData.append('file', this.FileInput.files[0])

http.onload = function() {
  console.log(http.responseText)
}

http.onerror = function() {
  console.error(http.responseText)
}

http.send(formData)

To insert file tugas siswa, you need to POST call to the following URL :
http://api.lms-sman5bdg.online/tugas/insert-assignment 



Result example 200:

{
  status: true,
  msg: 'File Tugas berhasil diupload'
}
                

                  
Result Example 500
{
  status: false,
  msg: 'Terdapat error'
}

POST QUERY PARAMETERS

Field Type Description
Header Authorization (mandatory) String bearer Your TOKEN.
Body tugas (mandatory) Integer Id tugas
Body file (mandatory) Blog / File Object JS Input File to upload
